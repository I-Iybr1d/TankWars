Angular install:

//Ir buscar os files de start ao git
git clone https://github.com/angular/quickstart.git [nome do projecto]
// entrar na pasta criada
cd [nome do projecto]
// istalar os modulos
npm install
// testar o projecto
npm start

three.js install:
// installar o threejs
npm install three-js
//Typescript definitions
npm install --save @types/three
